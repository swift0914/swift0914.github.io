# 函数类型

//函数声明
function foo(x: number, y: number): number {
    return x * y;
}
let result = foo(2, 3);
//函数表达式 此写法值对等号右侧匿名函数进行类型定义  不够完善
let foo2 = function(x: number, y: number): number {
    return x * y;
};
// typescript => 和es6中的有区别 typescript中=>表示函数的定义 左侧为类型 右侧表示返回类型
let foo3: (x: number, y: number) => number = (x: number, y: number): number => {
    return x * y;
};

// 接口的方式定义函数
interface searchFun {
    (target: string, subString: string): boolean;
}
let sFun: searchFun = function(target: string, subString: string): boolean {
    return target.search(subString) !== -1;
};

// 可选参数 必须放在必须参数后面
let name: (firstName: string, lastName?: string) => string = function(
    firstName: string = "sun", // 默认参数
    lastName?: string
): string {
    if (lastName) return firstName + " " + lastName;
    return firstName;
};

// 剩余参数  必须放参数末尾
let restFun: (arr: any[], ...items: any[]) => Array<any> = function(
    arr: any[],
    ...items: any[]
) {
    items.forEach((item) => {
        arr.push(item);
    });
    return arr;
};
console.log(restFun([], 1, "2", "5"));

# 类型断言

// 用途1 将联合类型断言为其中一个类型
interface Fish {
    name: string;
    swim(): void;
}
interface Cat {
    name: string;
    run(): void;
}
let isFish: (animal: Cat | Fish) => boolean = function(
    animal: Cat | Fish
): boolean {
    if (typeof (animal as Fish).swim === "function") return true;
    return false;
};

# 类型别名

通常使用在联合类型 使用 type 定义

type Name = string;
type NameFun = () => string;
type nameOrFun = Name | NameFun;
let foo: (n: nameOrFun) => Name = (n: nameOrFun): Name => {
    if (typeof n === "string") return n;
    return n();
};
console.log(
    foo(() => {
        return "123";
    })
);

# class

class Animal {
    // name: "";
    get name() {
        return "tom";
    }
    set name(value) {
        console.log("setVal " + value);
    }
    constructor(name) {
        this.name = name;
    }
    say() {
        // console.log(`my name is ${this.name}`);
        return `my name is ${this.name}`;
    }
    static isAnimal(obj) {
        // 类的静态方法
        return obj instanceof Animal;
    }
}
let cat = new Animal("tom");
cat.name = "cat";
console.log(cat.say());
console.log("staic --- " + Animal.isAnimal(cat));
try {
    cat.isAnimal(cat);
} catch (e) {
    console.warn(e);
}

// 类的继承
class Cat extends Animal {
    constructor(name) {
        super(name);
    }
    say() {
        // console.log("Meow, " + super.say());
        return "Meow, " + super.say();
    }
}
let tom = new Cat("tom");
console.log(tom.say());

// public private protected  装饰器
// class Person {
//     private name; // 私有属性  类的外部不可使用
//     protected gender; // 子类中允许被访问
//     constructor(name, gender) {
//         this.name = name;
//         this.gender = gender;
//     }
// }
class Person {
    //参数属性等同于上 更加简洁
    constructor(public name, public gender) {}
}
let swift = new Person("swift", "male");
console.log(swift.name);
swift.name = "jerry";
console.log(swift.name);

class Male extends Person {
    // protected constructor(name, gender) {  构造函数 protected 类只能被继承不可实例化  protected 类只能实例化不可被继承
    //     super(name, gender);
    // }
    constructor(name, gender) {
        super(name, gender);
    }
    say() {
        console.log(this.name);
        console.log("protected 子类中可以访问" + this.gender);
    }
}
let tony = new Male("tony", "male");
tony.say();

# 类实现接口

interface Alarm {
    alert(): void;
}
interface Light {
    lightOn(): void;
    lightOff(): void;
}
class Door {}
class SecurityDoor extends Door implements Alarm {
    alert() {
        console.log("SecurityDoor alert");
    }
}
class Car implements Alarm, Light {
    // 一个类可以实现多个接口
    alert() {
        console.log("Car alert");
    }
    lightOn() {
        console.log("Car light on");
    }
    lightOff() {
        console.log("Car light off");
    }
}

# 泛型

不先预定类型 使用的时候在指定

function arr(time: number, val: any): Array<any> {
    let result = [];
    for (let i = 0; i < time; i++) {
        result.push(val);
    }
    return result;
}

// 使用断言
function arr2<T>(time: number, val: T): T[] {
    let result: T[] = [];
    for (let i = 0; i < time; i++) {
        result.push(val);
    }
    return result;
}
