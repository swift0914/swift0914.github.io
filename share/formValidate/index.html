# 简单实现 Element 表单及验证功能

TIP

表单是前端非常常见的功能，表单验证也是必不可少的。 通过既定的规则验证用户输入内容，并友好提示校验结果，从而得到更好的用户体验。 主流的 web UI Element 在表单验证功能非常完善。此教程仅封装 Input 的校验功能，如对其他 form（select、textarea 等）组件感兴趣可自行研究。

# 1、Element 表单结构

<template>
  <el-form :model="form" :rules="rules" ref="form">
    <el-form-item label="用户名" prop="username">
      <el-input v-model="form.username" placeholder="请输入用户名" />
    </el-form-item>
    <el-form-item label="密码" prop="password">
      <el-input
        v-model="form.password"
        type="password"
        placeholder="请输入密码"
      />
    </el-form-item>
    <el-form-item>
      <el-button @click="submitForm('form')">登录</el-button>
    </el-form-item>
  </el-form>
</template>
<script>
submitForm(formName) {
  this.$refs[formName].validate((valid) => {
    if (valid) {
      alert('submit!');
    } else {
      console.log('error submit!!');
      return false;
    }
  });
}
</script>

# 1. FormInput 的实现

主要功能

  • input 类型的绑定
  • 自定义组件双向数据绑定的实现
  • 组件不作为 props 的 attribues 通过 v-bind="$attrs" 传递给 input(placeholder)
<template>
  <div>
    <!-- 
      1、input 类型
      2、自定义组件双向数据绑定的实现
      3、组件不作为 props 的attribues 通过 v-bind="$attrs" 传递给input
    -->
    <input :type="type" :value="value" @input="handleInput" v-bind="$attrs" />
  </div>
</template>

<script>
export default {
  inheritAttrs: false, // 去除根元素绑定非 props 的 attribues
  props: {
    type: {
      type: String,
      default: "text",
    },
    value: {
      type: String,
    },
  },
  methods: {
    handleInput(e) {
      this.$emit("input", e.target.value);
    },
  },
};
</script>

# 3、校验功能

# 1. 校验开源库 async-validator 介绍

async-validatoropen in new window

import Schema from "async-validator";
const descriptor = {
  // 校验的表述
  name: [
    {
      type: "string",
      required: true,
      message: "名称不能为空",
    },
    {
      min: 3,
      max: 5,
      message: "长度在 3 到 5 个字符",
    },
  ],
  age: [{
    type: "number",
    asyncValidator: (rule, value) => {
      return new Promise((resolve, reject) => {
        if (value < 18) {
          rule.message = "少儿不宜";
          reject("too young"); // reject with error message
        } else {
          resolve();
        }
      });
    },
  }],
};
const validator = new Schema(descriptor);
/** validate(source, [options], callback): Promise
 *  source: 需要校验的对象（必选）
 *  options: 描述校验的处理选项的对象（可选）
 *  callback: 当校验完成时调用的回调函数（必选）
 */
// 同步校验
validator.validate({ name: "" }, (errors) => {
  if (errors) {
    // 校验不通过 do something
    console.log(errors[0].message);
    return;
  }
  // 校验通过 do something
});
// 异步校验
validator
  .validate({ name: "swift", age: 16 })
  .then()
  .catch(({ errors }) => {
    console.log(errors[0].message);
  });

# 2. 何时触发校验、何处执行校验

  • input 触发 input 事件就应触发校验

  • 点击提交按钮触发校验

  • 校验的提示及方法在 FormItem 组件内部（组件间的低耦合 复用性）

# Form 改造

组建通信

model rules 的值在 Form 组件中，需要传递给 FormItem 作校验时使用，而 Form 组件中只有 slot，无法使用 prop 进行组件同行。 使用 provide/inject 进行跨层级通信。

# 添加功能
  • 组件跨层通信
  • 提供 validate 方法供提交按钮使用
+ provide() {
+   return {
+     form: this
+   }
+ }

+ methods: {
+   validate(cb) { // 供提交按钮使用 触发所有校验函数 返回一个boolean 所有校验通过 true 否则 false
+     // 注意判断一下子组件是否有 prop 提交按钮的 FormItem 无校验
+     const result = this.$children.reduce((prev, curr) => curr.prop ? prev.concat(curr.invalide()) : prev, []).every(item => item)
+     cb(result)
+   }
+ }

# FormItem 改造

添加功能

  • 接受 Form 组件传递的 model、rules
  • 添加校验函数
  • 监听 FormInput 触发的校验方法
+ import Schema from 'async-validator';
+ inject: ['form']
+ mounted() {
+   this.$on('invalide', () => this.invalide())
+ },
+ methods: {
+   invalide() {
+     const { model, rules } = this.form // 当前表单值 和 校验的规则
+     const desc = { // 校验描述
+       [this.prop]: rules[this.prop]
+     }
+     const validator = new Schema(desc)
+     // validate 返回的 promise return供 Form的validate函数使用
+     validator.validate(({ [this.prop]: model[this.prop] }), (errors) => {
+       this.error = errors ? errors[0].message : ''
+     })
+     return this.error ? false : true
+   }
+ }

# 4、不足及终极方案

不足之处

Form 中使用了 this.$children FormInput中使用了 this.$parent 组件之间耦合。

# 最终方案

使用 dispatch 的方式，派发事件，从而降低组件的耦合

// emitter.js mixins混入
export default {
  methods: {
    /**
     * 事件派发（自下而上） input => formItem  input 通知 formItem 触发检验事件
     *                   formItem => form   每创建一个formItem 通知 form 收集一次 formItem 组件实例
     * @param {*} componentName 通知的组件
     * @param {*} eventName 派发的时间名称
     * @param {*} params 额外参数
     */
    dispatch(componentName, eventName, params) {
      var parent = this.$parent || this.$root;
      var name = parent.$options.componentName;

      while (parent && (!name || name !== componentName)) {
        parent = parent.$parent;

        if (parent) {
          name = parent.$options.componentName;
        }
      }
      if (parent) {
        parent.$emit.apply(parent, [eventName].concat(params));
      }
    },
  },
};
