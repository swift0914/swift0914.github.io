# call、apply的区别

功能相似，区别在于参数的传递方式不同

# call、apply实现

# call

var name = 'catalina'
var obj = {
  name: 'swift'
}
function bar(job, age) {
  console.log(this.name)
  console.log(job, age);
}
Function.prototype.call = function (context) {
  /**
   * 第一个参数为null、undefined时 this指向到window
   * 第一个参数不是null、undefined时  必须是个对象
  **/
  if (!context) context = window
  context.fn = this
  let rest = [...arguments].slice(1)
  let result = context.fn(...rest)
  delete context.fn
  return result
}
bar.call(obj, 'IT', 18) // swift
bar.call(null, 'teacher', 18) // catalina

Function.prototype.apply = function (context, rest) {
  /**
   * 第一个参数为null、undefined时 this指向到window
   * 第一个参数不是null、undefined时  必须是个对象
  **/
  if (!context) context = window
  context.fn = this
  let result
  if (rest === null || rest === undefined) {
    result = context.fn(rest)
