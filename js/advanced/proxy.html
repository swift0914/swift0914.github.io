# 双向绑定

# 实现双向绑定 Proxy 和 Object.definePrototype 相比的优劣。

  • 1、Object.definePrototype 只可以劫持对象的某个属性，劫持属性的getter和setter方法，再对象属性发生变化时进行特定的操作。Proxy 可以劫持整个对象。
  • 2、Proxy 返回一个被劫持的全新的对象，我们处理新对象就可以，而 Object.definePrototype 是在原有对象上直接修改。
  • 3、Object.definePrototype 不支持对数组的劫持。
  • 4、Object.definePrototype 兼容性优于 Proxy。

# Object.definePrototype 实现双向绑定

var obj = {}, temp = "swift";
Object.defineProperty(obj, 'name', {
  get() {
    console.log('取值成功')
    return temp
  },
  set(value) {
    console.log('设置值成功')
    temp = value
  }
})
obj.name = '蔡徐坤' // 设置值成功
console.log(obj.name)  // 取值成功

# Proxy 实现双向绑定

注意

proxy 会劫持整个对象，当读取或者修改某个属性值的时候，都会被劫持到。需要注意的是，复杂类型只是监听了引用地址，而不是值，如果引用地址没有改变，是不会触发 set 的。

var obj = {
  name: '蔡徐坤',
  hobbits: ['chang', 'jump', 'rap'],
  info: {
    age: 3
  }
}
var p = new Proxy(obj, {
  get(target, key) {
    console.log('proxy 取值成功')
    return Reflect.get(target, key)
  },
  set(target, key, val) {
    if (key === length) return true // 如果是数组的长度 直接返回
    console.log('proxy 设置值成功')
    return Reflect.set(target, key, val)
  }
})
p.name = 'swift' // 触发 set
console.log(p.name) // swift
p.info.age = '222' // 不会触发 set
p.hobbits.push('photography') // 不会触发 set
console.log(p.info.age) // 222

# 对于数组的劫持

Object.definePrototype 可以将数组的索引作为属性进行劫持，但是仅仅支持 array[i] 进行才做，不支持数组的任何方法。

var arr = [], temp;
Object.defineProperty(arr, '0', {
  get() {
    console.log('取值成功')
    return temp
  },
  set(val) {
    console.log('设置值成功')
    temp = val
  }
})
arr[0] = 1 // 设置值成功
arr.push(2) // 不会被劫持

Proxy 可以监听到数组的变化，支持数组的方法。需要注意的是数组的变化触发 get 和 set可能不止一次，如果需要可以根据 key 来判断。

var arr = ['chang', 'tiao', 'rap']
var p = new Proxy(arr, {
  get(target, key) {
    console.log('取值成功')
    return Reflect.get(target, key)
  },
  set(target, key, val) {
    console.log('设置值成功')
    return Reflect.set(target, key, val)
  }
})
// p.splice(0, 1) // 触发了get和set
p.push('chi') // 触发get和set
p.slice(0) // 触发get slice不会改变原数组
