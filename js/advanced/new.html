# 手写实现 new

new 运算符创建了一个用户定义的某种对象类型的实例，或者具有构造函数的内置对象类型之一。

# 举个例子

function Coder(name) {
  this.name = name;
  this.habit = "coding";
}
Coder.prototype.strength = 30;
Coder.prototype.say = function () {
  console.log("name,", this.name);
};

var coder = new Coder("swift");
console.log(coder.name); // swift
console.log(coder.habit); // coding
console.log(coder.strength); // 30
coder.say();

# 实现功能

  • ①coder 实例可以访问到 构造函数内部的属性。
  • ②coder 实例可以访问到 构造函数原型上的属性和方法。

# 初步实现 new

function newFactory() {
  var obj = new Object(); // 创建空对象
  var Consturctor = [].shift.call(arguments); // 参数第一个为构造函数
  obj.__proto__ = Consturctor.prototype; // 实例 __proto__ 属性指向构造函数的原型对象上  ②
  Consturctor.apply(obj, arguments); // ①
  return obj;
}

# 完善 - 构造函数有返回值的情况

function newFactory() {
  var obj = new Object(); // 创建空对象
  var Consturctor = [].shift.call(arguments); // 参数第一个为构造函数
  obj.__proto__ = Consturctor.prototype; // 实例 __proto__ 属性指向构造函数的原型对象上  ②
  var res = Consturctor.apply(obj, arguments); // ①
  return typeof res === "object" ? res : obj;
}
