# 深浅拷贝、合并

# 浅拷贝

let shallowCopy = (obj) => {
    if (typeof obj !== "object") return;
    let newObj = obj instanceof Array ? [] : {};
    for (let key in obj) {
        if (obj.hasOwnProperty(key)) {
            newObj[key] = obj[key];
        }
    }
    return newObj;
};
let target = { a: "1", b: { c: "2" }, d: function () {} };
let shallowCopyObj = shallowCopy(target);
shallowCopyObj["b"]["c"] = "2222"; // target[b][c] 也会随之改变

# 深拷贝

let deepCopy = (obj) => {
    if (typeof obj !== "object") return;
    let newObj = obj instanceof Array ? [] : {};
    for (let key in obj) {
        if (obj.hasOwnProperty(key)) {
            newObj[key] =
                typeof obj[key] === "object" ? deepCopy(obj[key]) : obj[key];
        }
    }
    return newObj;
};
let target2 = { a: "1", b: { c: "2" }, d: function () {} };
let deepCopyObj = deepCopy(target2);
deepCopyObj["b"]["c"] = "2222"; // target2[b][c] 不会随之改变

# 合并对象

function extend() {
    // 所有值赋值给目标对象  浅拷贝
    let target = arguments[0]; // 目标对象
    let len = arguments.length;
    for (let i = 1; i < len; i++) {
        // 所有需要赋值给target
        if (arguments[i] != null) {
            // 确保传递了参数
            for (let key in arguments[i]) {
                target[key] = arguments[i][key];
            }
        }
    }
    return target;
}

function deepExtend() {
    let deep = false,
        i = 1,
        target = arguments[0] || {},
        len = arguments.length;
    if (typeof arguments[0] === "boolean") {
        // 判断第一个参数是否 boolean
        deep = true;
        target = arguments[i];
        i++;
    }
    for (; i < len; i++) {
        if (arguments[i] != null) {
            for (let key in arguments[i]) {
                if (deep && typeof arguments[i] == "object") {
                    target[key] = deepExtend(
                        deep,
                        target[key],
                        arguments[i][key]
                    );
                } else {
                    target[key] = arguments[i][key];
                }
            }
        }
    }
    return target;
}
let deepExtendobj = deepExtend(true, { a: 1 }, { a: 2, b: { c: 3 } });

let extendObj = extend({ a: 1 }, { a: 2, b: 3 }, { c: null });
