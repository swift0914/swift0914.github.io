# ES5 继承

# 1、原型链继承

缺点

  • 无法传递参数
  • 所有子类的原型都指向父类的实例对象 某个子类实例修改父类原型引用成员 所有子类实例被影响
function Parent() {
    this.attr = { name: "swift" };
}
Parent.prototype.sayHello = function() {
    console.log("hello, " + this.attr.name);
};
function Child() {}
Child.prototype = new Parent();
let child1 = new Child();
let child2 = new Child();
child1.sayHello();
child1.attr.name = "蔡徐坤";
child2.sayHello(); // 蔡徐坤

# 2、构造函数继承

缺点

  • 子类无法使用父类原型的属性和方法
function Parent(name) {
    this.name = name;
}
Parent.prototype.sayHello = function() {
    console.log("hello," + this.name);
};
function Child(name) {
    Parent.call(this, name);
}
let child = new Child("swift");
let child2 = new Child("蔡徐坤");
console.log(child.name);
console.log(child2.name);
try {
    child.sayHello();
} catch {
    console.error("构造函数继承 子类无法调用父类原型的方法和属性");
}

# 3、组合式继承

缺点

  • 每个子类实例都执行两次构造函数 Parent.call() new Parent()
function Parent(name) {
    this.name = name;
}
Parent.prototype.sayHello = function() {
    console.log("hello," + this.name);
};
function Child(name) {
    Parent.call(this, name); // 构造函数继承
}
Child.prototype = new Parent(); // 原型继承
let child = new Child("swift");
let child2 = new Child("蔡徐坤");
child.sayHello();
child2.sayHello();

# 4、寄生式组合继承

最为优的方案

function Parent(name) {
    this.name = name;
}
Parent.prototype.sayHello = function() {
    console.log("hello," + this.name);
};
function Child(name) {
    Parent.call(this, name); // 构造函数继承
}
Child.prototype = Object.create(Parent.prototype); // 浅拷贝 防止修改子类原型的属性、方法时影响到父类原型
Child.prototype.getName = function() {
    console.log("子类的原型上的方法:" + this.name);
};
const child = new Child("swift");
const parent = new Parent("super");
child.sayHello();
child.getName();
parent.sayHello();
try {
    parent.getName();
} catch {
    console.error("不使用Object.create()父类原型也会有getName方法");
}
