# Step-By-Step

# 1、如何正确的判断this的指向。

this 用一句话可以理解为 谁调用了它，this就指向谁。但是仅通过这句话有时还是无法准确的判断出this。this 的指向可以通过一下顺序进行判断。

# 1.全局中的this

浏览器环境下无论严格模式或非严格模式，this都指向 window。

# 2.是否有new绑定

如果存在new绑定，this就指向new的对象。

function Parent(name) {
  console.log(this)
  this.name = name
}
let instance = new Parent('cxk')

# 6.箭头函数，this指向上层上下文中的this

let obj = {
  name: 'cxk',
  say: function () {
    return () => {
      console.log(this)
    }
  }
}
let say = obj.say()
say() // obj

let person = { name: 'swift' }
let say1 = obj.say.call(person)
say1() // person

# 2、如何使 a==1&&1==2&&a==3 成立

== 操作符左右两边类型不同 需要隐式类型转换。a==1&&1==2&&a==3 意味着不可能是基础类型。因此a只能是复杂类型 object

# valueOf

var a = {
  valueOf: (function () {
    var i = 1;
    // 利用闭包  i无法释放
    return () => {
      console.log(i);
      return i++
    }
  })()
}
console.log(a == 1 && a == 2 && a == 3);

# hash

function unique(arr) {
  let newArr = [], hash = {}, len = arr.length;
  if (len === 0 || len === 1) return arr
  for (let i = 0; i < len; i++) {
    if (!hash[arr[i]]) {
      hash[arr[i]] = true;
      newArr.push(arr[i])
    }
  }
  return newArr
}

# Set

function unique(arr) {
  const len = arr.length
  if (len === 0 || len === 1) return arr
  return [...new Set(arr)]
}
